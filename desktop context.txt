━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1) Frontend quality system added and shown per response
  - Added live quality cards under every assistant response in Streamlit.
  - Shows heuristic signals:
      accuracy, relevance, groundedness, safety, latency, overall
  - Added formal overlap metrics display:
      precision, recall, F1, ROUGE-1, ROUGE-2, ROUGE-L, BLEU
  - Added threshold bands in UI:
      each metric shows "min X | great Y" plus status label.

2) Formal metric basis clarified
  - Metrics currently compare answer vs retrieved context (when available).
  - These are grounding-overlap indicators, NOT gold-answer correctness metrics.
  - Do not present them as clinical accuracy validation.

3) Retrieval/path improvements in backend
  - Retrieval no longer returns plain dense top-k only.
  - Implemented over-fetch + rerank:
      dense similarity + lexical overlap hybrid score
  - New config controls:
      retrieval_candidate_pool, retrieval_keyword_weight, retrieval_dense_weight

4) Prompting and output-structure tightening
  - Reworked RAG prompts into mode-specific templates:
      concise, detailed, structured
  - Prompts explicitly prohibit unsupported facts and require missing-info honesty.
  - Structured mode enforces sectioned output for higher coverage consistency.
  - Report tool and drug tool prompts also tightened and structured.

5) Frontend control improvements
  - Added "structured" answer mode in sidebar.
  - Added retrieval profiles:
      high_precision (Tight Evidence), balanced, high_recall (Broad Coverage)
  - Profiles auto-adjust top_k defaults.

6) LLM decoding controls moved to config
  - llm_temperature and llm_top_p now configurable (instead of hardcoded).
  - Current defaults set lower for hallucination control.

7) Chunking logic updated (future ingestion)
  - Updated chunker toward smaller chunks and higher overlap:
      chunk_size 350, overlap 75
  - Added sentence-aware windowing fallback behavior.
  - Critical caveat:
      this affects only newly ingested vectors.
      Existing Qdrant points remain old chunk boundaries until re-ingestion.
